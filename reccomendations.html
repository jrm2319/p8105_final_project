rec_data <- merge(books_clean, rate_clean, by = "book_id", all = TRUE)
authors <- unique(rec_data$authors)

genres <- unique(rec_data$top_1)


ui <- fluidPage(
  titlePanel("Book Recommendations"),
  
  sidebarLayout(
    sidebarPanel(
      selectizeInput("authors", 
                     label = "Start typing an author's name:", 
                     choices = authors, 
                     options = list(
                       placeholder = 'Select an author...',
                       create = FALSE, 
                       maxItems = 1,
                       highlight = TRUE)),
      
      selectInput("top_1", "Choose Genre:", 
                  choices = unique(rec_data$top_1), selected = NULL),
      
      selectInput("rating", "Choose Ratings:", 
                  choices = unique(rec_data$rating), selected = NULL),
      
      actionButton("recommend", "Get Recommendations")
    ),
    
    mainPanel(
      h3("Recommended Books"),
      DT::dataTableOutput("book_table")
    )
  )
)
## Warning: The select input "authors" contains a large number of options;
## consider using server-side selectize for massively improved performance. See
## the Details section of the ?selectizeInput help topic.
server <- function(input, output, session) {
  
  output$selected_author <- renderText({
    paste("You selected:", input$authors)
  })
  

  filtered_books <- reactive({
    rec_data %>%
      filter(
        (authors == input$authors | input$authors == "") &
        (top_1 == input$top_1 | input$top_1 == "") &
        (rating == input$rating | input$rating == "")
      )
  })
  

  observeEvent(input$recommend, {
    output$book_table <- DT::renderDataTable({
      filtered_books()
    })
  })
}


shinyApp(ui = ui, server = server)
